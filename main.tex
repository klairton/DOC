\NewEnviron{task}{
  \problemtitle{}\probleminput{}\problemtask{}% Default input is empty
  \BODY
  \begin{center}
  \fbox{
    \begin{minipage}{0.9\linewidth}
      \raggedright
      \parskip=1ex
      \textbf{\@problemtitle}          \\% Title
      \textbf{Entrada:} \@probleminput \\% Input
      \textbf{Objetivo:} \@problemtask \\% Goal
    \end{minipage}
  } 
  \end{center}
}
